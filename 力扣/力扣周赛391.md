# [第三题 交替子数组计数 - 力扣（LeetCode）](https://leetcode.cn/problems/count-alternating-subarrays/description/)
## 比赛我的做法
发现了若符合条件的子数组有n个元素，则其有$(1+n) * n / 2$个子数组，所以直接一次遍历找到不符合的地方，然后套公式计算，之后更新左端点，继续遍历即可
代码如下：
```cpp
class Solution {
public:
    long long countAlternatingSubarrays(vector<int>& nums) {
        long long ans = 0;
        int n = nums.size();
        int pos = 0;
        if(n != 1){
            for(int i = 1;i<n;i++){
                if(nums[i] == nums[i-1]){
                    ans = ans + (long long)(i - pos) * (long long)(1 + i - pos) / 2;
                    pos = i;
                }
            }
            if(nums[n-1] == nums[n-2]){
                ans++;
            }
            else{
                ans = ans + (long long)(n - pos) * (long long)(1 + n - pos) / 2;
            }
        }
        else if(n == 1){
            ans = 1;
        }
        return ans;
    }
};
```
注意考虑$n == 1$的情况
## 灵神做法
符合条件的子数组每多一个元素（假设已有n个元素），就会多n+1个子数组（从新加入的元素往前
数即可），是我的做法的本质简化版
```cpp
class Solution {
public:
    long long countAlternatingSubarrays(vector<int> &nums) {
        long long ans = 0, cnt = 0;
        for (int i = 0; i < nums.size(); i++) {
            i && nums[i] == nums[i - 1] ? cnt = 1 : cnt++;
            ans += cnt; // 有 cnt 个以 i 为右端点的交替子数组
        }
        return ans;
    }
};
```

# [第四题 最小化曼哈顿距离 - 力扣（LeetCode）](https://leetcode.cn/problems/minimize-manhattan-distances/description/)
## 有序集合
技巧：曼哈顿距离转切比雪夫距离，可以参照[距离 - OI Wiki (oi-wiki.org)](https://oi-wiki.org/geometry/distance/)
直接上结论：
$$(x', y') = (x + y, y - x)$$
$$\lvert x_1 - x_2 \rvert + \lvert y_1 - y_2 \rvert = \max \{\lvert x_1' - x_2' \rvert, \lvert y_1' - y_2' \rvert\}$$
然后接下来就转换成了删除某个点后，最大距离的最小值了，用一个有序集合维护所有坐标，每次去掉一个点的相应坐标，然后计算去min即可
```cpp
class Solution {
public:
    int minimumDistance(vector<vector<int>>& points) {
        multiset<int> xs,ys;
        for(auto &p : points){
            xs.insert(p[0] + p[1]);
            ys.insert(p[1] - p[0]);
        }
        int ans = INT_MAX;
        for(auto &p : points){
            int xs1 = p[0] + p[1];
            int ys1 = p[1] - p[0];
            xs.erase(xs.find(xs1));
            ys.erase(ys.find(ys1));
            ans = min(max(*xs.rbegin() - *xs.begin(), *ys.rbegin() - *ys.begin()), ans);
            xs.insert(xs1);
            ys.insert(ys1);
        }
        return ans;
    }
}; 
```
## 维护最大次大、最小次小
当去掉最大的时候，次大就变成最大，最小同理。
```cpp
class Solution {
    // 更新最大次大
    void update_max(int v, int &max1, int &max2) {
        if (v > max1) {
            max2 = max1;
            max1 = v;
        } else if (v > max2) {
            max2 = v;
        }
    }

    // 更新最小次小
    void update_min(int v, int &min1, int &min2) {
        if (v < min1) {
            min2 = min1;
            min1 = v;
        } else if (v < min2) {
            min2 = v;
        }
    }

public:
    int minimumDistance(vector<vector<int>> &points) {
        int max_x1 = INT_MIN, max_x2 = INT_MIN, max_y1 = INT_MIN, max_y2 = INT_MIN;
        int min_x1 = INT_MAX, min_x2 = INT_MAX, min_y1 = INT_MAX, min_y2 = INT_MAX;

        for (auto &p : points) {
            int x = p[0] + p[1];
            int y = p[1] - p[0];
            update_max(x, max_x1, max_x2);
            update_min(x, min_x1, min_x2);
            update_max(y, max_y1, max_y2);
            update_min(y, min_y1, min_y2);
        }

        int ans = INT_MAX;
        for (auto &p : points) {
            int x = p[0] + p[1];
            int y = p[1] - p[0];
            int dx = (x == max_x1 ? max_x2 : max_x1) - (x == min_x1 ? min_x2 : min_x1);
            int dy = (y == max_y1 ? max_y2 : max_y1) - (y == min_y1 ? min_y2 : min_y1);
            ans = min(ans, max(dx, dy));
        }
        return ans;
    }
};
```
