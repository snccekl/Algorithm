## 格式：
bool operator 运算符 (const 结构体名称 结构体) const{
    return (什么时候此运算符对结构体成立)//对此运算符使用this->元素名
}
```c++
#include <bits/stdc++.h>
using namespace std;
struct node{
	int id;
	int value;
	bool operator < (const node &x) const{
		return value < x.value;
	}
}a[100];
int main(){
	int n;
	cin>>n;
	for(int i = 0;i < n;i++){
		cin>>a[i].id>>a[i].value;
	}
	sort(a,a + n);
	for(int i = 0;i < n;i++){
		printf("%d %d",a[i].id,a[i].value);
	}
	return 0;
}
```
