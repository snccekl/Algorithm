# [P3741 honoka的键盘 - 洛谷 ](https://www.luogu.com.cn/problem/P3741)
## 方法1
第一种方法自然是暴力，考虑一个都不改和每一个改一次的可能性。时间复杂度显然是O（n^2）
## 方法2
先遍历连续两字母为VK的情况，在遍历过程中，遇到此情况便使结果加1，同时将之更改，避免接下来重复遍历（我选择更改为vk)
再遍历则只考虑连续两字母相同的情况，若有则加1返回即可。
**注意长度为1的情况**
```c++
#include <bits/stdc++.h>
using namespace std;
int main(){
    int l;
    string a;
    scanf("%d",&l);
    cin>>a;
    int ans = 0;
    if(a.size() == 1){
        printf("0");
        return 0;
    }
    for(int i = 0;i<a.size()-1;i++){
        if(a[i] == 'V' && a[i+1] == 'K'){
            ans++; 
            a[i] = 'v';
            a[i+1] = 'k';   
        }
    }
    for(int i = 0;i<a.size()-1;i++){
        if(a[i] == a[i+1]){
            ans++;
            break;
        }
    }
    printf("%d",ans);
    return 0;
}
```
## 补充
### cin>>（scanf类似）
用法1：输入一个数字或字符
用法2：接收一个字符串，遇“空格”、“TAB”、“回车”就结束
### cin.getline()
用法:接收一个字符串，可以接收空格并输出
扩展：  
1. cin.getline()实际上有三个参数，cin.getline(接收字符串的变量,接收字符个数,**结束字符**)  
2. 当第三个参数省略时，系统默认为`'\0'`  
### getline()
用法：接收一个字符串，可以接收空格并输出，需包含`string`头文件
### 注意
1. cin.getline()属于istream流，而getline()属于string流，是不一样的两个函数
2. 当同时使用cin>>（scanf）,getline()时，在cin>>输入流完成之后，getline()之前，必须要将回车符作为输入流cin以清除缓存，否则将在终端跳过getline()。因为程序会默认把之前的变量作为输入流

```cpp
#include<bits/stdc++.h>
using namespace std;
int n,m,a[505][505],c[505][505],tot;
void RotateS(int x,int y,int r)//顺时针
{
    if(r <= 1) return;
    int t=r/2,now;
    now=x-t+1;
    for(int i=y-t+1;i<=y+t;i++)//top
    {
        c[now][y+t]=a[x-t][i];
        now++;
    }
    now=y-t+1;
    for(int i=x+t-1;i>=x-t;i--)//left
    {
        c[x-t][now]=a[i][y-t];
        now++;
    }
    now=x+t-1;
    for(int i=y+t-1;i>=y-t;i--)//bottom
    {
        c[now][y-t]=a[x+t][i];
        now--;
    }
    now=y+t-1;
    for(int i=x-t+1;i<=x+t;i++)//right
    {
        c[x+t][now]=a[i][y+t];
        now--;
    }
    RotateS(x,y,r-2);//递归
}
void RotateN(int x,int y,int r)//逆时针
{
    if(r<=1) return;
    int t=r/2,now;
    now=x+t-1;
    for(int i=y-t+1;i<=y+t;i++)//top
    {
        c[now][y-t]=a[x-t][i];
        now--;
    }
    now=y+t-1;
    for(int i=x+t-1;i>=x-t;i--)//left
    {
        c[x+t][now]=a[i][y-t];
        now--;
    }
    now=x-t+1;
    for(int i=y+t-1;i>=y-t;i--)//bottom
    {
        c[now][y+t]=a[x+t][i];
        now++;
    }
    now=y-t+1;
    for(int i=x-t+1;i<=x+t;i++)//right
    {
        c[x-t][now]=a[i][y+t];
        now++;
    }
    RotateN(x,y,r-2);//递归
}
int main()
{
    scanf("%d %d",&m,&n);
    for(int i=1;i<=m;i++){
        for(int j=1;j<=m;j++){ 
	        a[i][j] = ++tot;
	    }

    for(int i=1;i<=m;i++){
        int x,y,r,z;
        scanf("%d %d %d %d",&x,&y,&r,&z);
        if(z==0){

            for(int i=x-r;i<=x+r;i++){
                for(int j=y-r;j<=y+r;j++){ 
	                a[i][j]=c[i][j];
	            }
            }

			    }
            }
```

# [P1518两只塔姆沃斯牛 ](https://www.luogu.com.cn/problem/P1518)
每分钟，农夫和牛可以向前移动或是转弯。
1. 如果前方无障碍（地图边沿也是障碍），它们会按照原来的方向前进一步。
2. 否则它们会用这一分钟顺时针转 90 度。
终止条件：两者相遇，并且在该分钟末尾。然后计算时间。

可以用一个变量ans存储分钟数，用一个二维字符串数组存储地图，地图大小为10 * 10，但为了便于判断是否越界，可以开大为12 * 12，将边界置为障碍$' * '$，

同时显然不能每次移动都遍历数组，太费时间，故可以用两个数组分别记录农夫与牛的位置和方向信息，之后只需要更改这些信息即可
## 关键在于判断是否相遇
怎么判断呢？我们可以想到，如果两个物体先后两次从同一个方向走到同一个地点，我们就可以说它们陷入了死循环，但如何判断是否是死循环？
可以通过生成专属值的方法，农夫的x坐标+他的y坐标* 10+奶牛的x坐标* 100+奶牛的y坐标* 1000+农夫的方向* 10000+奶牛的方向* 40000（农夫方向最多为4）

下为总代码：
```cpp
#include <bits/stdc++.h>
using namespace std;
int f[3],c[3];//f为农夫，c为牛。[0]为方向，从北顺时针为0、1、2、3，[1]为横坐标，[2]为纵坐标
char m[12][12];//地图
int ans,val;//ans为分钟数，val为专属值
bool zt[160005];//记录专属值是否出现
void move(int des,int x,int y,int h){//des为方向，x、y为横纵坐标，h为类型，农夫为0，奶牛为1
    if(des == 0){
        if (m[x-1][y]=='*'){
            if (h==0){
                f[0]=1;
            }
            else{
                c[0]=1;
            }
        }
        else if (h==0){
            f[1]--;
        }
        else
            c[1]--;
    }
    else if(des == 1){
        if (m[x][y+1]=='*'){
            if (h==0){
                f[0]=2;
            }
            else{
                c[0]=2;
            }
        }
        else if (h==0){
            f[2]++;
        }
        else
            c[2]++;
    }
    else if(des == 2){
        if (m[x+1][y]=='*'){
            if (h==0){
                f[0]=3;
            }
            else{
                c[0]=3;
            }
        }
        else if (h==0){
            f[1]++;
        }
        else
            c[1]++;
    }
    else if(des == 3){
        if (m[x][y-1]=='*'){
            if (h==0){
                f[0]=0;
            }
            else{
                c[0]=0;
            }
        }
        else if (h==0){
            f[2]--;
        }
        else
            c[2]--;
    }
}
bool judge(){
    if(f[1] == c[1] && f[2] == c[2]){
        return false;
    }
    return true;
}
int main(){
    for (int i=0;i<=11;i++) {
        m[i][0]='*';
        m[i][11]='*';
    }
    for (int i=1;i<=11;i++){
        m[0][i]='*';
        m[11][i]='*';
    }
    for (int i=1;i<=10;i++){
        for (int j=1;j<=10;j++){
        cin>>m[i][j];
        if (m[i][j]=='F') f[1]=i,f[2]=j;
        if (m[i][j]=='C') c[1]=i,c[2]=j;
        }
    }
    while(judge()){
        val = f[1]+f[2]*10+c[1]*100+c[2]*1000+f[0]*10000+c[0]*40000;
        if(zt[val]){
            cout<<0;
            return 0;
        }
        zt[val] = true;
        move(f[0],f[1],f[2],0);
        move(c[0],c[1],c[2],1);
        ans++;
    }
    cout<<ans<<endl;
    return 0;
}
```

# [P1067 多项式输出 - 洛谷](https://www.luogu.com.cn/problem/P1067)
题目很简单，但是要注意的点很多

**1号坑**：开头如果为正则不输出符号
**2号坑**：系数为1或-1时只输出符号不输出系数
**3号坑**：系数为0时不输出该项
**4号坑**：一次项不要输出系数
下为代码实现
```cpp
#include<bits/stdc++.h>
using namespace std;
int main(){
    int n,a;
    cin>>n;
    for(int i=n;i>=0;i--){
        cin>>a;
        if(a){    判0系数
            if(i!=n&&a>0)cout<<"+";    根据正负、是否为最高此项决定加号
            if(abs(a)>1||i==0)cout<<a;    输出系数（系数不为正负1或指数为0）
            if(a==-1&&i)cout<<"-";    -1系数特判，常数项已特判
            if(i>1)cout<<"x^"<<i;    二次及以上输出指数
            if(i==1)cout<<"x";    一次项
        }
    }
}
```
